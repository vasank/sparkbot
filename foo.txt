/** Get Query Param value */
function getQeryParamValue(str,name) {
  name = name.replace(/[\[]/, "\\\[").replace(/[\]]/, "\\\]");
  var regexS = "[\\?&]" + name+ "=([^&#]*)";
  var regex = new RegExp(regexS);
  var results = regex.exec(str);
  if(results == null)
    return "";
  else
    return decodeURIComponent(results[1].replace(/\+/g, " "));
}
